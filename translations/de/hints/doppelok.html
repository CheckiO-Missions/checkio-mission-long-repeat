Deshalb werden wir das Argument "default" in der Funktion max() verwenden, wenn die Funktion den maximalwert nicht berechnen kann, gibt sie "0" zurück. 
def long_repeat(line: str) -> int:
    result = []
    lenline = len(line)
    while line:
        line = line.lstrip(line[0])
        result.append(lenline - len(line))
        lenline = len(line)
    return max(result, default=0)  # we return the maximum value
