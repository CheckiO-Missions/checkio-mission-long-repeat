Тому ми будемо використовувати аргумент "за замовчуванням" у функції max(), якщо функція не зможе обчислити максимальне значення, вона поверне "0". 
def long_repeat(line: str) -> int:
    result = []
    lenline = len(line)
    while line:
        line = line.lstrip(line[0])
        result.append(lenline - len(line))
        lenline = len(line)
    return max(result, default=0)  # we return the maximum value
