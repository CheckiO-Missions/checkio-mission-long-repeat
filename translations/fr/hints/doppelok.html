Par conséquent, nous utiliserons l'argument "default" dans la fonction max(), si la fonction ne peut pas calculer la valeur maximale, elle renverra "0" la valeur maximale, elle renverra "0". 
def long_repeat(line: str) -> int:
    result = []
    lenline = len(line)
    while line:
        line = line.lstrip(line[0])
        result.append(lenline - len(line))
        lenline = len(line)
    return max(result, default=0)  # we return the maximum value
